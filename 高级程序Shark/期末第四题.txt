#include <iostream>
#include <cmath>

using namespace std;
struct Fraction {
    int numerator;      // 分子
    int denominator;    // 分母
};

int gcd(int a, int b) {
    if (a % b == 0) {
        return b;
    }
    else {
        return gcd(b, a % b);
    }
}


Fraction simplify(Fraction f) {
   
    if (f.numerator == 0) {
        f.denominator = 1;
    }
   
    else if (f.denominator == 1) {
        // do nothing
    }
    
    else {
        int k = gcd(abs(f.numerator), abs(f.denominator));
        f.numerator /= k;
        f.denominator /= k;
    }
    return f;
}

int main() {
    int n;
    cout << "请输入分数个数：";
    cin >> n;

    Fraction sum = { 0, 1 };  

    // 读取n个分数并求和
    for (int i = 0; i < n; i++) {
        Fraction f;
        cout << "请输入第" << i + 1 << "个分数（格式为 分子/分母）：";
        scanf_s("%d/%d", &f.numerator, &f.denominator);
        // 将当前分数加入总和
        sum.numerator = sum.numerator * f.denominator + f.numerator * sum.denominator;
        sum.denominator *= f.denominator;
    }

    
    Fraction result = simplify(sum);
    cout << "分数之和为：" << result.numerator << "/" << result.denominator << endl;

    return 0;
}
